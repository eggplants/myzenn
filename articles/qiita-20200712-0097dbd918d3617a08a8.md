## プログラム

`out`というファイルに書き込む

```python:ope.py
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from time import sleep

o = Options()
o.add_argument('--headless')
o.add_argument('--disable-gpu')

d = webdriver.Chrome(
  options=o
)

first = 'https://www.amazon.co.jp/b?node=8187562051'
url = 'https://www.amazon.co.jp/'\
      's?rh=n%3A2250738051%2Cn%3A%212275265051%2Cn%3A'\
      '%212275277051%2Cn%3A%218172192051%2Cn%3A'\
      '%218187544051%2Cn%3A%218187545051%2Cn%3A'\
      '%218187561051%2Cn%3A8187562051'
first_xpath = '//h2[@data-attribute]'
title_xpath = '//span[@class = "a-size-medium a-color-base a-text-normal"][@dir="auto"]'

# 6分待つ
titles = set()
for i in range(1, 76+1):
  print('#Page %02d'%i, end="\r")
  if i != 1:
    d.get(url+('&page=%d'%i))
    titles |= {i.text for i in d.find_elements_by_xpath(title_xpath)}
  else:
    d.get(first)
    titles |= {i.text for i in d.find_elements_by_xpath(first_xpath)}
  sleep(5) # 十分にwait

f = open('out', 'w')
[print(i, file=f) for i in titles]

d.quit()
```
